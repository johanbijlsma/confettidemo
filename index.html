Confetti Example

Exploring Confetti options. in this example I'm looking at party.js


    let siteColors = ['#0A9F8E', '#E5F7F7', '#009EE0', '#0B6B6F'];

    party.element(document.querySelector('body'), {
        count: party.variation(300, 0.5),
        angleSpan: party.minmax(60, 120),
        spread: 360,
        size: 16,
        color: siteColors,
        lightning: true,
    });
